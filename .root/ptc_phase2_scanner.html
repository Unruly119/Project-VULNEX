HTTP Headers คืออะไร? — ก่อนเขียนโค้ด ต้องเข้าใจก่อน
แนวคิด ⏱ 5 นาที
concept
▶
ทุกครั้งที่เบราว์เซอร์ขอเว็บไซต์ เซิร์ฟเวอร์จะตอบกลับมา 2 ส่วนคือ "ส่วนหัว (Headers)" และ "เนื้อหา (Body)"

Headers คือข้อมูล metadata ที่ซ่อนอยู่ ผู้ใช้ทั่วไปมองไม่เห็น แต่ นักทดสอบความปลอดภัยใช้ดูว่าเว็บไซต์ตั้งค่าความปลอดภัยถูกต้องไหม
🛡️ Content-Security-Policy
ป้องกัน XSS Attack — กำหนดว่าเว็บโหลด script จากที่ไหนได้บ้าง ถ้าไม่มีเสี่ยงมาก
🔒 Strict-Transport-Security
บังคับให้ใช้ HTTPS เสมอ ป้องกัน Man-in-the-Middle Attack
🪟 X-Frame-Options
ป้องกัน Clickjacking — ไม่ให้คนฝังเว็บเราใน iframe ของเว็บอื่น
🔍 X-Content-Type-Options
ป้องกัน MIME Sniffing — บังคับเบราว์เซอร์อ่านไฟล์ตามประเภทที่กำหนด
💡
โปรเจกต์เราจะทำอะไร? ดึง Headers เหล่านี้จากเว็บไซต์เป้าหมาย แล้วส่งให้ AI วิเคราะห์ว่ามีหรือไม่มี ตั้งค่าถูกหรือผิด
ถ้าเว็บไซต์ไม่มี X-Frame-Options header จะเสี่ยงต่อการโจมตีแบบใด?
SQL Injection
Clickjacking
DDoS Attack
Brute Force
✅ ถูกต้อง! Clickjacking คือการที่คนร้ายซ่อนเว็บไซต์ไว้ใน iframe แล้วหลอกให้ผู้ใช้กดโดยไม่รู้ตัว X-Frame-Options: DENY จะป้องกันได้
❌ ไม่ใช่ครับ X-Frame-Options ป้องกัน Clickjacking — การซ่อนเว็บไว้ใน iframe

1# ── src/scanner/headers.py ──────────────────────
2# หน้าที่: ดึง HTTP Security Headers จากเว็บไซต์
3
4import httpx httpx = library ส่ง HTTP request
5from typing import Dict, Optional บอก Python ว่า return type คืออะไร
6
7# Security Headers ที่เราจะตรวจสอบ
8# เก็บเป็น list เพื่อให้เพิ่มได้ง่ายในอนาคต
9SECURITY_HEADERS = [ รายการที่ต้องตรวจ
10 "Content-Security-Policy",
11 "Strict-Transport-Security",
12 "X-Frame-Options",
13 "X-Content-Type-Options",
14 "Referrer-Policy",
15 "Permissions-Policy",
16]
17
18def check_headers(url: str) -> Dict: function หลัก
19 """
20 ดึง Security Headers จาก URL ที่ระบุ
21 Return: dict ที่บอกว่า header แต่ละตัวมีหรือไม่
22 """
23 result = { เก็บผลลัพธ์
24 "url": url,
25 "headers_found": {}, # headers ที่เจอ
26 "headers_missing": [], # headers ที่หายไป
27 "score": 0, # คะแนน 0-100
28 "error": None, # error message ถ้ามี
29 }
30
31 try: จัดการ error
32 # ส่ง HTTP request ไปยัง URL
33 # timeout=10 รอไม่เกิน 10 วินาที
34 # follow_redirects เพื่อตามไป https อัตโนมัติ
35 with httpx.Client(
36 timeout=10,
37 follow_redirects=True,
38 verify=False # ยอมรับ SSL cert ที่หมดอายุด้วย
39 ) as client:
40 response = client.head(url) HEAD = ขอแค่ headers ไม่ขอ body
41
42 # ตรวจสอบทีละ header
43 for header in SECURITY_HEADERS: วนลูปตรวจทีละตัว
44 if header.lower() in response.headers: .lower() = ไม่สนตัวพิมพ์
45 result["headers_found"][header] = response.headers[header.lower()]
46 else:
47 result["headers_missing"].append(header) ไม่มี = ช่องโหว่
48
49 # คำนวณคะแนน: กี่ % ของ headers ที่มี
50 found = len(result["headers_found"])
51 total = len(SECURITY_HEADERS)
52 result["score"] = round(found / total * 100) คะแนน 0-100
53
54 except Exception as e: ถ้าเกิด error
55 result["error"] = str(e)
56
57 return result ส่งผลลัพธ์กลับ
🧠
สิ่งที่ต้องจำจากโค้ดนี้ def = สร้าง function (กล่องใส่โค้ดที่ใช้ซ้ำได้)
try/except = ถ้าเกิด error จะไม่ crash ทั้งโปรแกรม
for...in = วนซ้ำทีละรายการ
dict[key] = เก็บและดึงข้อมูลแบบ key-value
ผลลัพธ์ที่ได้เมื่อทดสอบ
{ "url": "https://example.com", "headers_found": { "Strict-Transport-Security": "max-age=31536000", "X-Content-Type-Options": "nosniff" }, "headers_missing": [ "Content-Security-Policy", "X-Frame-Options", "Referrer-Policy", "Permissions-Policy" ], "score": 33, "error": null }

เขียน ssl_check.py — ตรวจ SSL Certificate
📄 src/scanner/ssl_check.py ⏱ 10 นาที
ไฟล์ใหม่ code
▶
SSL Certificate คือ "ใบรับรองตัวตน" ของเว็บไซต์ — ที่เห็น 🔒 ในเบราว์เซอร์ เราจะตรวจว่า cert ยังไม่หมดอายุ, ชื่อโดเมนตรงกัน, และ ใช้ algorithm ที่ปลอดภัย
src/scanner/ssl_check.py
1import ssl, socket มีใน Python ไม่ต้อง pip install
2from datetime import datetime
3from urllib.parse import urlparse แยก URL ออกเป็นส่วน ๆ
4
5def check_ssl(url: str) -> dict:
6 """ตรวจสอบ SSL Certificate ของ URL"""
7 result = {
8 "has_ssl": False, # มี https ไหม
9 "valid": False, # cert ยังใช้งานได้ไหม
10 "days_left": 0, # เหลืออีกกี่วัน
11 "issuer": "", # ใครออก cert (Let's Encrypt, DigiCert ฯลฯ)
12 "expires": "", # วันหมดอายุ
13 "warning": "", # คำเตือนถ้ามี
14 "error": None
15 }
16
17 # ตรวจว่าใช้ HTTPS ไหม
18 if not url.startswith("https://"):
19 result["warning"] = "เว็บไม่ได้ใช้ HTTPS — ข้อมูลไม่เข้ารหัส!"
20 return result หยุดทันที ถ้าไม่มี https
21
22 result["has_ssl"] = True
23
24 try:
25 # แยก hostname ออกจาก URL
26 # เช่น "https://school.ac.th/page" → "school.ac.th"
27 hostname = urlparse(url).hostname ดึงเฉพาะ domain name
28
29 # เชื่อมต่อ SSL และดึง certificate
30 ctx = ssl.create_default_context()
31 with socket.create_connection((hostname, 443), timeout=5) as sock:
32 with ctx.wrap_socket(sock, server_hostname=hostname) as ssock:
33 cert = ssock.getpeercert() ดึงข้อมูล certificate
34
35 # ดึงวันหมดอายุ
36 expire_str = cert["notAfter"] # เช่น "Dec 31 23:59:59 2025 GMT"
37 expire_date = datetime.strptime(expire_str, "%b %d %H:%M:%S %Y %Z")
38 days_left = (expire_date - datetime.utcnow()).days
39
40 result["expires"] = expire_date.strftime("%d/%m/%Y")
41 result["days_left"] = days_left
42 result["valid"] = days_left > 0
43
44 # เตือนถ้าจะหมดอายุใน 30 วัน
45 if 0 < days_left <= 30:
46 result["warning"] = f"⚠️ SSL จะหมดอายุใน {days_left} วัน!" f-string
47
48 # ดึงชื่อผู้ออก cert
49 issuer_dict = dict(x[0] for x in cert["issuer"])
50 result["issuer"] = issuer_dict.get("organizationName", "Unknown")
51
52 except Exception as e:
53 result["error"] = str(e)
54 result["valid"] = False
55
56 return result
💡
f-string คืออะไร? (บรรทัด 46)
f"ข้อความ {variable}" — ใส่ตัวแปรใน string โดยใช้ปีกกา เช่น f"เหลือ {30} วัน" → "เหลือ 30 วัน"

src/scanner/html_parser.py
1import httpx
2from bs4 import BeautifulSoup bs4 = BeautifulSoup4
3
4def parse_html(url: str) -> dict:
5 """ดึงและวิเคราะห์ HTML จากเว็บไซต์"""
6 result = {
7 "title": "", # ชื่อเว็บ
8 "meta_description": "", # คำอธิบายเว็บ
9 "external_scripts": [], # script จาก domain อื่น
10 "insecure_forms": [], # form ที่ส่งข้อมูลไป http
11 "total_links": 0, # จำนวน link ทั้งหมด
12 "error": None
13 }
14
15 try:
16 # GET request ดึง HTML ทั้งหน้า
17 resp = httpx.get(url, timeout=15, follow_redirects=True, verify=False)
18
19 # ให้ BeautifulSoup "อ่าน" HTML
20 # "lxml" คือ parser ที่เร็วที่สุด
21 soup = BeautifulSoup(resp.text, "lxml") soup = วัตถุที่ค้นหา HTML ได้
22
23 # ── ดึง title ──
24 title_tag = soup.find("title") หา <title> tag
25 if title_tag:
26 result["title"] = title_tag.text.strip()
27
28 # ── ดึง meta description ──
29 meta = soup.find("meta", {"name": "description"}) หา meta ที่ name=description
30 if meta:
31 result["meta_description"] = meta.get("content", "")
32
33 # ── ตรวจ external scripts (อันตราย!) ──
34 from urllib.parse import urlparse
35 base_domain = urlparse(url).netloc
36 for script in soup.find_all("script", src=True): หา script ที่มี src
37 src = script["src"]
38 if src.startswith("http") and base_domain not in src:
39 result["external_scripts"].append(src) script จาก domain อื่น
40
41 # ── ตรวจ forms ที่ action ไป http:// (ส่งข้อมูลไม่เข้ารหัส) ──
42 for form in soup.find_all("form"):
43 action = form.get("action", "")
44 if action.startswith("http://"): http ไม่ใช่ https = อันตราย
45 result["insecure_forms"].append(action)
46
47 # นับ links ทั้งหมด
48 result["total_links"] = len(soup.find_all("a", href=True))
49
50 except Exception as e:
51 result["error"] = str(e)
52
53 return result
